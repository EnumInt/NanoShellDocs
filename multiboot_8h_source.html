47/* Align all boot modules on i386 page (4KB) boundaries. */

56/* This flag indicates the use of the address fields in the header. */

59/* Flags to be set in the 'flags' member of the multiboot info structure. */

61/* is there basic lower/upper memory information? */

64#define MULTIBOOT_INFO_BOOTDEV 0x00000002
65/* is the command-line defined? */
66#define MULTIBOOT_INFO_CMDLINE 0x00000004

68#define MULTIBOOT_INFO_MODS 0x00000008

70/* These next two are mutually exclusive */

72/* is there a symbol table loaded? */

102struct multiboot_header
103{
104 /* Must be MULTIBOOT_MAGIC - see above. */
105 MultibootUInt32 magic;
106
107 /* Feature flags. */
108 MultibootUInt32 flags;
109
110 /* The above fields plus this one must equal 0 mod 2^32. */
111 MultibootUInt32 checksum;
112
113 /* These are only valid if MULTIBOOT_AOUT_KLUDGE is set. */
114 MultibootUInt32 header_addr;
115 MultibootUInt32 load_addr;
116 MultibootUInt32 load_end_addr;
117 MultibootUInt32 bss_end_addr;
118 MultibootUInt32 entry_addr;
119
120 /* These are only valid if MULTIBOOT_VIDEO_MODE is set. */
121 MultibootUInt32 mode_type;
122 MultibootUInt32 width;
123 MultibootUInt32 height;
124 MultibootUInt32 depth;
125};
