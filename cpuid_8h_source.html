181/* Return highest supported input value for cpuid instruction. ext can
182 be either 0x0 or 0x8000000 to return highest supported value for
183 basic or extended cpuid information. Function returns 0 if cpuid
184 is not supported or whatever cpuid returns in eax register. If sig
185 pointer is non-null, then first four bytes of the signature
186 (as found in ebx register) are returned in location pointed by sig. */
187
188static __inline unsigned int
189__get_cpuid_max (unsigned int __ext, unsigned int *__sig)
190{
191 unsigned int __eax, __ebx, __ecx, __edx;
192
193#ifndef __x86_64__
194 /* See if we can use cpuid. On AMD64 we always can. */
195#if __GNUC__ >= 3
196 __asm__ ("pushf{l|d}\n\t"
197 "pushf{l|d}\n\t"
198 "pop{l}\t%0\n\t"
199 "mov{l}\t{%0, %1|%1, %0}\n\t"
200 "xor{l}\t{%2, %0|%0, %2}\n\t"
201 "push{l}\t%0\n\t"
202 "popf{l|d}\n\t"
203 "pushf{l|d}\n\t"
204 "pop{l}\t%0\n\t"
205 "popf{l|d}\n\t"
206 : "=&r" (__eax), "=&r" (__ebx)
207 : "i" (0x00200000));
208#else
209/* Host GCCs older than 3.0 weren't supporting Intel asm syntax
210 nor alternatives in i386 code. */
211 __asm__ ("pushfl\n\t"
212 "pushfl\n\t"
213 "popl\t%0\n\t"
214 "movl\t%0, %1\n\t"
215 "xorl\t%2, %0\n\t"
216 "pushl\t%0\n\t"
217 "popfl\n\t"
218 "pushfl\n\t"
219 "popl\t%0\n\t"
220 "popfl\n\t"
221 : "=&r" (__eax), "=&r" (__ebx)
222 : "i" (0x00200000));
223#endif
224
225 if (!((__eax ^ __ebx) & 0x00200000))
226 return 0;
227#endif
228
229 /* Host supports cpuid. Return highest supported cpuid input value. */
230 __cpuid (__ext, __eax, __ebx, __ecx, __edx);
231
232 if (__sig)
233 *__sig = __ebx;
234
235 return __eax;
236}

238/* Return cpuid data for requested cpuid leaf, as found in returned
239 eax, ebx, ecx and edx registers. The function checks if cpuid is
240 supported and returns 1 for valid cpuid information or 0 for
241 unsupported cpuid leaf. All pointers are required to be non-null. */
242
243static __inline int
244__get_cpuid (unsigned int __leaf,
245 unsigned int *__eax, unsigned int *__ebx,
246 unsigned int *__ecx, unsigned int *__edx)
247{
248 unsigned int __ext = __leaf & 0x80000000;
249
250 if (__get_cpuid_max (__ext, 0) < __leaf)
251 return 0;
252
253 __cpuid (__leaf, *__eax, *__ebx, *__ecx, *__edx);
254 return 1;
255}
